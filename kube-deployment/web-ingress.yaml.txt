apiVersion: extensions/v1beta1
kind: Ingress
metadata:
  name: "web-ingress"
  namespace: "web"
  annotations:
    kubernetes.io/ingress.class: alb
    alb.ingress.kubernetes.io/scheme: internet-facing
    alb.ingress.kubernetes.io/target-type: ip
    alb.ingress.kubernetes.io/subnets: mysubnets
  labels:
    app: web-ingress
spec:
  rules:
    - http:
        paths:
          - path: /*
            backend:
              serviceName: "web"
              servicePort: 80
